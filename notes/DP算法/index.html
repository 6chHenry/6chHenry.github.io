    def minDistance(word1, word2):
    m, n = len(word1), len(word2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    # åˆå§‹åŒ–è¾¹ç•Œï¼šå˜ä¸ºç©ºå­—ç¬¦ä¸²çš„æ“ä½œæ¬¡æ•°
    for i in range(m + 1): dp[i][0] = i
    for j in range(n + 1): dp[0][j] = j

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if word1[i-1] == word2[j-1]:
                dp[i][j] = dp[i-1][j-1] # å­—ç¬¦ç›¸åŒï¼Œæ— æ“ä½œ
            else:
                # ä¸‰é€‰ä¸€ï¼šæ›¿æ¢ã€åˆ é™¤ã€æ’å…¥
                dp[i][j] = min(dp[i-1][j-1], dp[i-1][j], dp[i][j-1]) + 1

    # å›æº¯æ“ä½œè·¯å¾„
    steps = []
    i, j = m, n
    while i > 0 or j > 0:
        if i > 0 and j > 0 and word1[i-1] == word2[j-1]:
            steps.append(f"Keep '{word1[i-1]}'")
            i, j = i - 1, j - 1
        else:
            # çœ‹çœ‹å½“å‰ dp[i][j] æ˜¯ä»å“ªä¸ªæ–¹å‘æ¨è¿‡æ¥çš„
            if i > 0 and j > 0 and dp[i][j] == dp[i-1][j-1] + 1:
                steps.append(f"Replace '{word1[i-1]}' with '{word2[j-1]}'")
                i, j = i - 1, j - 1
            elif i > 0 and dp[i][j] == dp[i-1][j] + 1:
                steps.append(f"Delete '{word1[i-1]}'")
                i -= 1
            else:
                steps.append(f"Insert '{word2[j-1]}'")
                j -= 1

    return dp[m][n], steps[::-1]
